'use client';

import { useState, useEffect } from 'react';

export default function Accounts() {
  const [accounts, setAccounts] = useState<any[]>([]);
  const [loading, setLoading] = useState(true);
  const [error, setError] = useState<string | null>(null);

  useEffect(() => {
    async function fetchAccounts() {
      try {
        const response = await fetch('/api/accounts');
        
        if (!response.ok) {
          throw new Error(`HTTP error! status: ${response.status}`);
        }
        
        const data = await response.json();
        setAccounts(data);
      } catch (err) {
        setError(err instanceof Error ? err.message : 'Unknown error');
      } finally {
        setLoading(false);
      }
    }
    fetchAccounts();
  }, []);

  if (loading) return <div className="p-8 text-center">Зареждане...</div>;
  if (error) return <div className="p-8 text-center text-red-500">Грешка: {error}</div>;

  return (
    <div className="p-8">
      <h1 className="text-2xl font-bold mb-4">Сметкоплан</h1>
      <div className="space-y-2">
        {accounts.map((account) => (
          <div key={account.id} className="p-3 border rounded">
            <span className="font-mono font-bold">{account.code}</span> - {account.name}
          </div>
        ))}
      </div>
    </div>
  );
}
